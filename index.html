import cv2 import numpy as np from svgpathtools import Path, Line, wsvg def image_to_svg(image_path, output_svg_path): # Cargar la imagen y convertirla a escala de grises image = cv2.imread(image_path, cv2.IMREAD_GRAYSCALE) # Convertir la imagen a binaria (umbral) _, binary_image = cv2.threshold(image, 127, 255, cv2.THRESH_BINARY) # Encontrar contornos en la imagen binaria contours, _ = cv2.findContours(binary_image, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE) # Crear un objeto Path para el archivo SVG path = Path() # Recorrer cada contorno y convertirlo en l√≠neas de SVG for contour in contours: for i in range(len(contour) - 1): start_point = tuple(contour[i][0]) end_point = tuple(contour[i + 1][0]) path.append(Line(start=start_point, end=end_point)) # Guardar el archivo SVG wsvg(path, filename=output_svg_path) # Llamada de ejemplo image_to_svg('imagen_de_entrada.png', 'salida.svg')
